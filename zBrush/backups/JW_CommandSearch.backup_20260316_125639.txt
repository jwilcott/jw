[RoutineDef, JWSearch_ResetResults,
	[VarSet, jwSearchResultCount, 0]
	[VarSet, jwSearchPage, 0]
	[Loop, 128,
		[VarSet, jwSearchNames(jwSearchLoopIndex), ""]
		[VarSet, jwSearchPaths(jwSearchLoopIndex), ""]
		[VarSet, jwSearchKinds(jwSearchLoopIndex), ""]
		[VarSet, jwSearchDescriptions(jwSearchLoopIndex), ""]
	, jwSearchLoopIndex]
]

[RoutineDef, JWSearch_AddResult,
	[If, jwSearchResultCount < 128,
		[VarSet, jwSearchNames(jwSearchResultCount), jwEntryName]
		[VarSet, jwSearchPaths(jwSearchResultCount), jwEntryPath]
		[VarSet, jwSearchKinds(jwSearchResultCount), jwEntryKind]
		[VarSet, jwSearchDescriptions(jwSearchResultCount), jwEntryDescription]
		[VarInc, jwSearchResultCount]
	]
, jwEntryName, jwEntryPath, jwEntryKind, jwEntryDescription]

[RoutineDef, JWSearch_MatchEntry,
	[VarSet, jwSearchHaystack, [StrLower, [StrMerge, jwEntryName, " ", jwEntryKeywords, " ", jwEntryDescription]]]
	[If, [StrFind, jwSearchQuery, jwSearchHaystack] >= 0,
		[RoutineCall, JWSearch_AddResult, jwEntryName, jwEntryPath, jwEntryKind, jwEntryDescription]
	]
, jwEntryName, jwEntryPath, jwEntryKind, jwEntryDescription, jwEntryKeywords]

[RoutineDef, JWSearch_ExecuteResult,
	[If, [IExists, jwSearchPaths(jwSearchSelectedIndex)],
		[IPress, jwSearchPaths(jwSearchSelectedIndex)]
	,
		[Note, [StrMerge, "This item path was not found in your current ZBrush UI:\n", jwSearchPaths(jwSearchSelectedIndex), "\n\nSelected item: ", jwSearchNames(jwSearchSelectedIndex), "\n\nThat usually means the button name differs in your build or the item is not currently available."], , 0]
	]
]

[RoutineDef, JWSearch_ShowResults,
	[VarSet, jwSearchStart, (jwSearchPage * jwSearchPageSize)]
	[VarSet, jwSearchVisibleCount, 0]
	[VarSet, jwSearchHasPrev, 0]
	[VarSet, jwSearchHasNext, 0]
	[VarSet, jwSearchNoteText, [StrMerge, "JW Command Search\nSearch: ", jwSearchQuery, "\nMatches: ", jwSearchResultCount, "\n\nChoose a result to run it."]]

	[If, jwSearchResultCount <= 0,
		[NoteIButton, "Search Again"]
		[NoteIButton, "Close"]
		[VarSet, jwSearchChoice, [Note, [StrMerge, "No matches for: ", jwSearchQuery, "\n\nTry short terms like trim, move, mask, dyna, zremesh, subtool, symmetry, clay, or polish."], , 0]]
		[If, jwSearchChoice == 1,
			[RoutineCall, JWSearch_Run]
		]
		[Exit]
	]

	[If, jwSearchPage > 0,
		[VarSet, jwSearchHasPrev, 1]
	]

	[Loop, jwSearchPageSize,
		[VarSet, jwSearchCurrentIndex, (jwSearchStart + jwSearchSlot)]
		[If, jwSearchCurrentIndex >= jwSearchResultCount,
			[LoopExit]
		]
		[VarSet, jwSearchVisibleCount, (jwSearchVisibleCount + 1)]
		[NoteIButton, [StrMerge, jwSearchNames(jwSearchCurrentIndex), " [", jwSearchKinds(jwSearchCurrentIndex), "]"]]
		[VarSet, jwSearchNoteText, [StrMerge, jwSearchNoteText, "\n\n", jwSearchNames(jwSearchCurrentIndex), " [", jwSearchKinds(jwSearchCurrentIndex), "]\n", jwSearchDescriptions(jwSearchCurrentIndex)]]
	, jwSearchSlot]

	[If, (jwSearchStart + jwSearchVisibleCount) < jwSearchResultCount,
		[VarSet, jwSearchHasNext, 1]
	]

	[If, jwSearchHasPrev == 1,
		[NoteIButton, "Prev Page"]
	]
	[NoteIButton, "Search Again"]
	[If, jwSearchHasNext == 1,
		[NoteIButton, "Next Page"]
	]
	[NoteIButton, "Close"]

	[VarSet, jwSearchChoice, [Note, jwSearchNoteText, , 0, 0x2d2d2d, , 680]]

	[If, jwSearchChoice <= 0,
		[Exit]
	]

	[If, jwSearchChoice <= jwSearchVisibleCount,
		[VarSet, jwSearchSelectedIndex, (jwSearchStart + jwSearchChoice - 1)]
		[RoutineCall, JWSearch_ExecuteResult]
	]

	[VarSet, jwSearchExtraButtonIndex, (jwSearchVisibleCount + 1)]
	[If, jwSearchHasPrev == 1,
		[If, jwSearchChoice == jwSearchExtraButtonIndex,
			[VarDec, jwSearchPage]
			[RoutineCall, JWSearch_ShowResults]
			[Exit]
		]
		[VarInc, jwSearchExtraButtonIndex]
	]

	[If, jwSearchChoice == jwSearchExtraButtonIndex,
		[RoutineCall, JWSearch_Run]
		[Exit]
	]
	[VarInc, jwSearchExtraButtonIndex]

	[If, jwSearchHasNext == 1,
		[If, jwSearchChoice == jwSearchExtraButtonIndex,
			[VarInc, jwSearchPage]
			[RoutineCall, JWSearch_ShowResults]
			[Exit]
		]
	]
]

[RoutineDef, JWSearch_Run,
	[VarSet, jwSearchRawQuery, [StrAsk, jwSearchQuery, "Search ZBrush Commands and Brushes"]]
	[If, [StrLength, jwSearchRawQuery] <= 0,
		[Exit]
	]
	[VarSet, jwSearchQuery, [StrLower, jwSearchRawQuery]]
	[RoutineCall, JWSearch_ResetResults]
	[RoutineCall, JWSearch_BuildCatalog]
	[RoutineCall, JWSearch_ShowResults]
]

[IButton, ???, "Search ZBrush commands and brushes. Assign a hotkey to this macro after it loads.", [RoutineCall, JWSearch_Run]]
